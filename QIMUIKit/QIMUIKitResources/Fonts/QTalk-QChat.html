第一步：使用font-face声明字体 
  @font-face {font-family: 'QTalk-QChat';
    src: url('QTalk-QChat.eot'); /* IE9*/
    src: url('QTalk-QChat.eot?#iefix') format('embedded-opentype'), /* IE6-IE8 */
    url('QTalk-QChat.woff') format('woff'), /* chrome、firefox */
    url('QTalk-QChat.ttf') format('truetype'), /* chrome、firefox、opera、Safari, Android, iOS 4.2+*/
    url('QTalk-QChat.svg#iconfont') format('svg'); /* iOS 4.1- */
  }
第二步：定义使用iconfont的样式 
  .iconfont {
    font-family:"QTalk-QChat" !important;
    font-size:16px;font-style:normal;
    -webkit-font-smoothing: antialiased;
    -webkit-text-stroke-width: 0.2px;
    -moz-osx-font-smoothing: grayscale;}
